    @property

        """Number of latent variables."""
        # return number of traits in BV matrix plus 1
        return 1

    @property
    def C(self) -> numpy.ndarray:
        """Cholesky decomposition of the kinship matrix."""
        return self._C

    def latentfn(
            self, 
            x: numpy.ndarray, 
            *args: tuple, 
            **kwargs: dict
        ) -> numpy.ndarray:
        """
        Score a subset of individuals based according to its mean expected 
        heterozygosity (MEH).
        
        Scoring for MEH is defined as the negative MEH for a selected subset. 
        A smaller value indicates a better MEH score.

        Parameters
        ----------
        x : numpy.ndarray
            A candidate solution vector of shape ``(k,) == (ndecn,)``.
        args : tuple
            Additional non-keyword arguments.
        kwargs : dict
            Additional keyword arguments.
        
        Returns
        -------
        out : numpy.ndarray
            A matrix of shape ``(1,)`` containing MEH values.
        """
        # calculate MEH
        # (n,n)[:,(k,)] -> (n,k)
        # (1/k) * (n,k).sum(1) -> (n,)
        Cx = (1.0 / len(x)) * self._C[:,x].sum(1)

        # norm2( (n,) ) -> (1,)
        dist = numpy.linalg.norm(Cx, ord = 2, keepdims = True)

        # calculate negative MEH
        # scalar - (1,) -> (1,)
        out = -(1.0 - dist)

        return out

    def latentfn(
            self, 
            x: numpy.ndarray, 
            *args: tuple, 
            **kwargs: dict
        ) -> numpy.ndarray:
        """
        Score a subset of individuals based according to its mean expected 
        heterozygosity (MEH).
        
        Scoring for MEH is defined as the negative MEH for a selected subset. 
        A smaller value indicates a better MEH score.

        Parameters
        ----------
        x : numpy.ndarray
            A candidate solution vector of shape ``(s,) == (ndecn,) == (ntaxa,)``.
            On entry, this vector is scaled to have a unit sum, such that
            ``latentfn(x) == latentfn(ax)`` where ``a`` is any number.
        args : tuple
            Additional non-keyword arguments.
        kwargs : dict
            Additional keyword arguments.
        
        Returns
        -------
        out : numpy.ndarray
            A matrix of shape ``(1,)`` containing MEH values.
        """
        # calculate sum(x)
        xsum = x.sum()

        # if sum(x) ~== 0, then set to 1
        xsum = xsum if abs(xsum) >= 1e-10 else 1.0

        # scale x to have a sum of 1 (contribution)
        # (n,) -> (n,)
        contrib = (1.0 / xsum) * x

        # calculate negated MEH
        # (n,n) . (n,) -> (n,)
        # norm( (n,) ) -> (1,)
        # scalar - (1,) -> (1,)
        out = -(1.0 - numpy.linalg.norm(self._C.dot(contrib), ord = 2, keepdims = True))

        return out

    def latentfn(
            self, 
            x: numpy.ndarray, 
            *args: tuple, 
            **kwargs: dict
        ) -> numpy.ndarray:
        """
        Score a subset of individuals based according to its mean expected 
        heterozygosity (MEH).
        
        Scoring for MEH is defined as the negative MEH for a selected subset. 
        A smaller value indicates a better MEH score.

        Parameters
        ----------
        x : numpy.ndarray
            A candidate solution vector of shape ``(k,) == (ndecn,)``.
        args : tuple
            Additional non-keyword arguments.
        kwargs : dict
            Additional keyword arguments.
        
        Returns
        -------
        out : numpy.ndarray
            A matrix of shape ``(1,)`` containing MEH values.
        """
        # calculate sum(x)
        xsum = x.sum()

        # if sum(x) ~== 0, then set to 1
        xsum = xsum if abs(xsum) >= 1e-10 else 1.0

        # scale x to have a sum of 1 (contribution)
        # (n,) -> (n,)
        contrib = (1.0 / xsum) * x

        # calculate negated MEH
        # (n,n) . (n,) -> (n,)
        # norm( (n,) ) -> (1,)
        # scalar - (1,) -> (1,)
        out = -(1.0 - numpy.linalg.norm(self._C.dot(contrib), ord = 2, keepdims = True))

        return out
